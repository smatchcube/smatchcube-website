#include <stdio.h>

int main(void) /* do some calculations on toes */

	int twice_toes = 2 * toes;
	int square_toes = toes * toes;

	printf("toes = %d, twice toes = %d, squared toes = %d\n", toes, twice_toes, square_toes);

	return 0;
}
