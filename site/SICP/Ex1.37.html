(define (cont-frac n d k)

  1. Our previous cont-frac procedure generates a recursive process, here is the iterative version of cont-frac.
(define (cont-frac n d k)
  (define (iter result k)
    (if (< k 1)
        result
        (iter (/ (n k) (+ (d k) result)) (- k 1))))
  (iter 0 k))
